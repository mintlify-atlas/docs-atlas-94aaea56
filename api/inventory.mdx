## Overview

The Inventory API provides comprehensive inventory and materials management including stock tracking, requisitions, material catalog, and warehouse operations.

**Base Path:** `/inventarios/`

## Material Search

### List Materials

Retrieve materials with pagination and filtering.

```http
GET /inventarios/api/materials/
```

<ParamField query="q" type="string">
  Search query to match against name, SKU, or description
</ParamField>

<ParamField query="category" type="integer">
  Filter by category ID
</ParamField>

<ParamField query="ubicacion" type="integer">
  Filter by warehouse location ID (only shows materials with stock in that location)
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

#### Response

<ResponseField name="results" type="array">
  Array of material objects (24 per page)
  <Expandable title="Material Object">
    <ResponseField name="id" type="integer">
      Material ID
    </ResponseField>
    <ResponseField name="nombre" type="string">
      Material name
    </ResponseField>
    <ResponseField name="sku" type="string">
      Stock Keeping Unit code
    </ResponseField>
    <ResponseField name="unidad" type="string">
      Unit of measure (e.g., "Unit", "Meter", "Kilogram")
    </ResponseField>
    <ResponseField name="precio_estimado" type="float">
      Estimated price per unit
    </ResponseField>
    <ResponseField name="stock" type="float">
      Current stock quantity
    </ResponseField>
    <ResponseField name="categoria" type="string">
      Category name
    </ResponseField>
    <ResponseField name="image_url" type="string">
      Material image URL or default SVG
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="has_next" type="boolean">
  Whether there are more pages
</ResponseField>

<ResponseField name="num_pages" type="integer">
  Total number of pages
</ResponseField>

<ResponseField name="current_page" type="integer">
  Current page number
</ResponseField>

<CodeGroup>

```javascript Request
fetch('/inventarios/api/materials/?q=bearing&category=5&page=1', {
  credentials: 'same-origin'
})
.then(res => res.json())
.then(data => console.log(data));
```

```json Response
{
  "results": [
    {
      "id": 234,
      "nombre": "SKF Ball Bearing 6205-2RS",
      "sku": "BRG-6205-2RS",
      "unidad": "Unit",
      "precio_estimado": 25.50,
      "stock": 45.0,
      "categoria": "Bearings",
      "image_url": "https://softcom.ccg.hn/media-proxy/inventarios/bearing_234.jpg"
    }
  ],
  "has_next": false,
  "num_pages": 1,
  "current_page": 1
}
```

</CodeGroup>

<ResponseField name="sku" type="string">

<ResponseField name="stock_total" type="float">

</ResponseField>

<ResponseField name="ubicaciones" type="array">

### Get Material by SKU

Search for a material by its SKU code (for barcode scanner).

```http
GET /inventarios/api/sku/?sku={sku_code}
```

<ParamField query="sku" type="string" required>
  SKU code to search
</ParamField>

#### Response

<ResponseField name="found" type="boolean">
  Whether material was found
</ResponseField>

<ResponseField name="material" type="object">
  Material details (if found)
</ResponseField>

<CodeGroup>

```javascript Request
fetch('/inventarios/api/sku/?sku=BRG-6205-2RS', {
  credentials: 'same-origin'
})
.then(res => res.json())
.then(data => {
  if (data.found) {
    console.log('Material found:', data.material);
  }
});
```

```json Response
{
  "found": true,
  "material": {
    "id": 234,
    "nombre": "SKF Ball Bearing 6205-2RS",
    "sku": "BRG-6205-2RS",
    "stock": 45.0,
    "precio_estimado": 25.50,
    "unidad": "Unit",
    "categoria": "Bearings",
    "image_url": "/media-proxy/inventarios/bearing_234.jpg"
  }
}
```

</CodeGroup>

## Shopping Cart

### Add to Cart

Add materials to the requisition cart.

```http
POST /inventarios/cart/add/
```

<ParamField body="material_id" type="integer" required>
  Material ID to add
</ParamField>

<ParamField body="cantidad" type="float" required>
  Quantity to request
</ParamField>

<ParamField body="ubicacion_id" type="integer">
  Preferred warehouse location
</ParamField>

<ParamField body="nota" type="string">
  Optional note for this item
</ParamField>

#### Response

<ResponseField name="status" type="string">
  "success" or "error"
</ResponseField>

<ResponseField name="message" type="string">
  Status message
</ResponseField>

<ResponseField name="cart_count" type="integer">
  Total items in cart
</ResponseField>

<CodeGroup>

```javascript Request
fetch('/inventarios/cart/add/', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-CSRFToken': csrfToken
  },
  credentials: 'same-origin',
  body: JSON.stringify({
    material_id: 234,
    cantidad: 2,
    nota: 'For pump P-101 maintenance'
  })
});
```

```json Response
{
  "status": "success",
  "message": "Material added to cart",
  "cart_count": 3
}
```

</CodeGroup>

### Submit Cart (Checkout)

Create a requisition from cart items.

```http
POST /inventarios/cart/checkout/
```

<ParamField body="justificacion" type="string" required>
  Requisition justification/reason
</ParamField>

<ParamField body="prioridad" type="string" default="NORMAL">
  Priority level: BAJA, NORMAL, ALTA, URGENTE
</ParamField>

<ParamField body="fecha_requerida" type="string">
  Required date (YYYY-MM-DD)
</ParamField>

<ParamField body="proyecto_id" type="integer">
  Associated project ID
</ParamField>

<ParamField body="ot_id" type="integer">
  Associated work order ID
</ParamField>

#### Response

<ResponseField name="status" type="string">
  "success" or "error"
</ResponseField>

<ResponseField name="solicitud_id" type="integer">
  Created requisition ID
</ResponseField>

<ResponseField name="mensaje" type="string">
  Status message
</ResponseField>

<CodeGroup>

```javascript Request
fetch('/inventarios/cart/checkout/', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-CSRFToken': csrfToken
  },
  credentials: 'same-origin',
  body: JSON.stringify({
    justificacion: 'Bearings needed for scheduled pump maintenance',
    prioridad: 'ALTA',
    fecha_requerida: '2024-03-20',
    ot_id: 1523
  })
});
```

```json Response
{
  "status": "success",
  "solicitud_id": 789,
  "mensaje": "Requisition created successfully"
}
```

</CodeGroup>

### Dispatch Requisition

Process and dispatch materials from warehouse.

```http
POST /inventarios/api/pedidos/{pk}/despachar/
```

<ParamField path="pk" type="integer" required>
  Requisition ID
</ParamField>

<ParamField body="items" type="array" required>
  Array of dispatch items
  <Expandable title="Dispatch Item">
    <ParamField body="item_id" type="integer" required>
      Requisition item ID
    </ParamField>
    <ParamField body="cantidad" type="float" required>
      Quantity to dispatch
    </ParamField>
    <ParamField body="lote_id" type="integer">
      Specific lot/batch ID
    </ParamField>
    <ParamField body="ubicacion_id" type="integer" required>
      Warehouse location to dispatch from
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="notas" type="string">
  Dispatch notes
</ParamField>

#### Response

<ResponseField name="status" type="string">
  "success" or "error"
</ResponseField>

<ResponseField name="message" type="string">
  Status message
</ResponseField>

<ResponseField name="movimientos" type="array">
  Array of created inventory movement IDs
</ResponseField>

<CodeGroup>

```javascript Request
fetch('/inventarios/api/pedidos/789/despachar/', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-CSRFToken': csrfToken
  },
  credentials: 'same-origin',
  body: JSON.stringify({
    items: [
      {
        item_id: 1001,
        cantidad: 2,
        ubicacion_id: 5
      }
    ],
    notas: 'Dispatched for OT-2024-1523'
  })
});
```

```json Response
{
  "status": "success",
  "message": "Requisition dispatched successfully",
  "movimientos": [5678, 5679]
}
```

</CodeGroup>
